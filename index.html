jQuery Selectors & Events

// CSS-style selectors
$('p') // All paragraphs
$('.class') // By class
$('#id') // By ID
$('div > p') // Child selector
$('[data-*]') // Attribute selector

// Event handling
$('#button').click(function() {
alert('Button clicked!');
});

DOM Manipulation with jQuery

// Content manipulation
$('#content').text('New text');
$('#content').html('<b>Bold text</b>');
$('#input').val('New value');

// CSS manipulation
$('.box').css('color', 'red');
$('.box').addClass('highlight');
$('.box').removeClass('old-class');

// DOM insertion
$('#container').append('<p>New paragraph</p>');
$('#container').prepend('<h3>Title</h3>');

Animations & Effects

// Basic animations
$('#box').hide(); // Instant hide
$('#box').fadeOut(); // Fade out
$('#box').slideUp(); // Slide up

// Custom animations
$('#box').animate({
left: '250px',
opacity: 0.5
}, 1000);
